Show & Hide Animations with bounce.js

Get bounce.js, first.

Example; 

new Noty({
    ...
    text: 'NOTY - a jquery notification library!',
    ...
    animation: {
        open: function () {

        }
        close: function () {
            var n = this;
            new mojs.Html({
                el        : n.barDom,
                x         : {0: 500, delay: 10, duration: 500, easing: 'cubic.out'},
                skewY     : {0: 10, delay: 10, duration: 500, easing: 'cubic.out'},
                isForce3d : true,
                onComplete: function () {
                    n.barDom.parentNode.removeChild(n.barDom); // removing noty's dom from document
                }
            }).play();
        }
    }
}).show();
Important: You need to remove NOTY's dom on close.
