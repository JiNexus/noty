Show & Hide Animations with velocity.js

Get velocity.js, first. 


new Noty({
    ...
    text: 'NOTY - animating with velocity!',
    ...
    animation: {
        open: function () {
            var n = this;
            Velocity(n.barDom, {
                left: 450,
                scaleY: 2
            }, {
                duration: 0
            });
            Velocity(n.barDom, {
                left: 0,
                scaleY: 1
            }, {
                easing: [ 8, 8 ]
            });
        },
        close: function () {
            var n = this;
            Velocity(n.barDom, {
                left: '+=-50'
            }, {
                easing: [ 8, 8, 2],
                duration: 350
            });
            Velocity(n.barDom, {
                left: 450,
                scaleY: .2,
                height: 0,
                margin: 0
            }, {
                easing: [ 8, 8 ],
                complete: function () {
                    n.barDom.parentNode.removeChild(n.barDom);
                }
            });
        }
    }
}).show();
Important: You need to remove NOTY's dom on close.
